CREATE TABLE vote_queue
(
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
first INT(6) UNSIGNED NOT NULL,
second INT(6) UNSIGNED NOT NULL
);


CREATE TABLE vote_cnt (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
first INT(6) UNSIGNED NOT NULL,
second INT(6) UNSIGNED NOT NULL,
count INT(6) UNSIGNED NOT NULL
)
